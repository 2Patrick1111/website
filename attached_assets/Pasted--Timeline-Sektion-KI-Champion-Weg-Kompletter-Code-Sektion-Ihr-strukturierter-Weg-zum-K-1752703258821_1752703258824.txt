## 3. JavaScript-Code (Timeline-Animationen)

```javascript
// Timeline Animation Controller
class TimelineController {
    constructor() {
        this.timelineItems = document.querySelectorAll('.timeline-item');
        this.timelineProgress = document.getElementById('timelineProgress');
        this.timelineDots = document.querySelectorAll('.timeline-dot');
        this.observer = null;
        this.init();
    }
    
    init() {
        if (this.timelineItems.length === 0) return;
        
        this.setupIntersectionObserver();
        this.addScrollListener();
        console.log('Timeline animations initialized');
    }
    
    setupIntersectionObserver() {
        const options = {
            root: null,
            rootMargin: '-20% 0px -20% 0px',
            threshold: 0.5
        };
        
        this.observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                const item = entry.target;
                const index = Array.from(this.timelineItems).indexOf(item);
                
                if (entry.isIntersecting) {
                    this.activateTimelineItem(item, index);
                } else {
                    this.deactivateTimelineItem(item, index);
                }
            });
        }, options);
        
        this.timelineItems.forEach(item => {
            this.observer.observe(item);
        });
    }
    
    activateTimelineItem(item, index) {
        item.classList.add('active');
        if (this.timelineDots[index]) {
            this.timelineDots[index].classList.add('active');
        }
        
        // Update progress line
        this.updateProgressLine(index + 1);
    }
    
    deactivateTimelineItem(item, index) {
        item.classList.remove('active');
        if (this.timelineDots[index]) {
            this.timelineDots[index].classList.remove('active');
        }
    }
    
    updateProgressLine(activeIndex) {
        if (!this.timelineProgress) return;
        
        const totalItems = this.timelineItems.length;
        const progressPercent = (activeIndex / totalItems) * 100;
        
        this.timelineProgress.style.height = `${Math.min(progressPercent, 100)}%`;
    }
    
    addScrollListener() {
        let ticking = false;
        
        const handleScroll = () => {
            if (!ticking) {
                requestAnimationFrame(() => {
                    this.updateTimelineProgress();
                    ticking = false;
                });
                ticking = true;
            }
        };
        
        window.addEventListener('scroll', handleScroll, { passive: true });
    }
    
    updateTimelineProgress() {
        if (!this.timelineProgress) return;
        
        const scrollTop = window.pageYOffset;
        const windowHeight = window.innerHeight;
        const documentHeight = document.documentElement.scrollHeight;
        
        // Calculate scroll progress
        const scrollPercent = scrollTop / (documentHeight - windowHeight);
        const clampedPercent = Math.max(0, Math.min(100, scrollPercent * 100));
        
        // Only update if timeline is in view
        const timelineSection = document.querySelector('.timeline-section');
        if (timelineSection) {
            const rect = timelineSection.getBoundingClientRect();
            if (rect.top <= windowHeight && rect.bottom >= 0) {
                this.timelineProgress.style.height = `${clampedPercent}%`;
            }
        }
    }
}

// Initialize timeline when DOM is ready
document.addEventListener('DOMContentLoaded', () => {
    // Wait a bit for all elements to be rendered
    setTimeout(() => {
        new TimelineController();
    }, 100);
});

console.log('Timeline controller loaded');
```
